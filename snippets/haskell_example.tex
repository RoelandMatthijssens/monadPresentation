\begin{minted}{haskell}
    isSubsequenceOf :: (Eq a) => [a] -> [a] -> Bool
    isSubsequenceOf []    _                    = True
    isSubsequenceOf _     []                   = False
    isSubsequenceOf a@(x:a') (y:b) | x == y    = isSubsequenceOf a' b
                                | otherwise = isSubsequenceOf a b
\end{minted}
